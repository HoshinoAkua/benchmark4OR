import json
import random

# Function to generate random float numbers within a specified range
def generate_random_float(min_value, max_value):
    return random.uniform(min_value, max_value)

# Number of reservoirs and fields
R = random.randint(5, 10)  # Randomly choose between 5 to 10 reservoirs
F = random.randint(5, 10)  # Randomly choose between 5 to 10 fields

# Generate data
data = {
    "reservoirs": [generate_random_float(1000, 10000) for _ in range(R)],
    "fields": [generate_random_float(500, 5000) for _ in range(F)],
    "costs": [[generate_random_float(1, 100) for _ in range(F)] for _ in range(R)],
    "quality": [generate_random_float(50, 100) for _ in range(R)],
    "min_quality": [generate_random_float(50, 100) for _ in range(F)]
}

# Save data to a JSON file
with open('data.json', 'w') as f:
    json.dump(data, f, indent=4)
