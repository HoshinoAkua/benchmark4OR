import json
import random

# Function to generate random float numbers
def generate_random_float(min_val, max_val, size):
    return [random.uniform(min_val, max_val) for _ in range(size)]

# Number of power plants and regions
num_plants = random.randint(5, 10)  # Randomly choose between 5 and 10 power plants
num_regions = random.randint(5, 10)  # Randomly choose between 5 and 10 regions

# Generate random data
supply = generate_random_float(1000, 10000, num_plants)
demand = generate_random_float(1000, 10000, num_regions)
cost = [[random.uniform(1, 100) for _ in range(num_regions)] for _ in range(num_plants)]
emission_rate = generate_random_float(0.1, 10, num_plants)
emission_limit = generate_random_float(1000, 10000, num_plants)

# Create a dictionary to store the data
data = {
    'supply': supply,
    'demand': demand,
    'cost': cost,
    'emission_rate': emission_rate,
    'emission_limit': emission_limit
}

# Save the data to a JSON file
with open('data.json', 'w') as f:
    json.dump(data, f, indent=4)
