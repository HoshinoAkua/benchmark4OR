import json
import random

# Function to generate random float numbers within a specified range
def generate_random_floats(size, lower_bound, upper_bound):
    return [random.uniform(lower_bound, upper_bound) for _ in range(size)]

# Define the number of power plants, regions, and renewable sources
num_plants = random.randint(3, 10)
num_regions = random.randint(3, 10)
num_renewables = random.randint(3, 10)

# Generate random data
supply = generate_random_floats(num_plants, 1000, 10000)
demand = generate_random_floats(num_regions, 1000, 10000)
cost = [[random.uniform(1, 100) for _ in range(num_regions)] for _ in range(num_plants)]
renewable_supply = generate_random_floats(num_renewables, 500, 5000)
penalty_cost = generate_random_floats(num_regions, 50, 500)

# Create the data dictionary
data = {
    'supply': supply,
    'demand': demand,
    'cost': cost,
    'renewable_supply': renewable_supply,
    'penalty_cost': penalty_cost
}

# Save the data to a JSON file
with open('data.json', 'w') as f:
    json.dump(data, f, indent=4)
