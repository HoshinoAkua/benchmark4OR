import json
import random

# Function to generate random float numbers within a specified range
def generate_random_float(min_val, max_val):
    return random.uniform(min_val, max_val)

# Number of reservoirs and fields
num_reservoirs = random.randint(5, 10)  # Randomly choose between 5 and 10 reservoirs
num_fields = random.randint(5, 10)  # Randomly choose between 5 and 10 fields

# Generate random data
reservoirs = [generate_random_float(1000, 10000) for _ in range(num_reservoirs)]
fields = [generate_random_float(500, 5000) for _ in range(num_fields)]
costs = [[generate_random_float(1, 100) for _ in range(num_fields)] for _ in range(num_reservoirs)]

# Create data dictionary
data = {
    'reservoirs': reservoirs,
    'fields': fields,
    'costs': costs
}

# Save data to 'data.json'
with open('data.json', 'w') as f:
    json.dump(data, f, indent=4)
