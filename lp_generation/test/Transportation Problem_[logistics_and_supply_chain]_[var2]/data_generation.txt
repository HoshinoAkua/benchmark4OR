import json
import random

# Function to generate random float numbers within a specified range
def generate_random_float(min_value, max_value):
    return random.uniform(min_value, max_value)

# Number of warehouses and stores
num_warehouses = random.randint(3, 10)  # Randomly choose between 3 and 10 warehouses
num_stores = random.randint(3, 10)  # Randomly choose between 3 and 10 stores

# Generate random supply for each warehouse
supply = [generate_random_float(1000, 10000) for _ in range(num_warehouses)]

# Generate random demand for each store
demand = [generate_random_float(1000, 10000) for _ in range(num_stores)]

# Generate random transportation cost from each warehouse to each store
cost = [[generate_random_float(1, 100) for _ in range(num_stores)] for _ in range(num_warehouses)]

# Generate random capacity for each route from warehouse to store
capacity = [[generate_random_float(500, 5000) for _ in range(num_stores)] for _ in range(num_warehouses)]

# Create a dictionary to hold the data
data = {
    'supply': supply,
    'demand': demand,
    'cost': cost,
    'capacity': capacity
}

# Save the data to a JSON file
with open('data.json', 'w') as f:
    json.dump(data, f, indent=4)
