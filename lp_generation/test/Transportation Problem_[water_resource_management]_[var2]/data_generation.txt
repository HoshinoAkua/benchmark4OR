import json
import random

def generate_data():
    # Number of reservoirs and fields
    R = random.randint(5, 10)
    F = random.randint(5, 10)
    
    # Generate random data
    reservoirs = [random.uniform(1000, 10000) for _ in range(R)]
    fields = [random.uniform(500, 5000) for _ in range(F)]
    costs = [[random.uniform(1, 100) for _ in range(F)] for _ in range(R)]
    evaporation_rates = [[random.uniform(0.01, 0.1) for _ in range(F)] for _ in range(R)]
    
    # Create data dictionary
    data = {
        'reservoirs': reservoirs,
        'fields': fields,
        'costs': costs,
        'evaporation_rates': evaporation_rates
    }
    
    # Save to data.json
    with open('data.json', 'w') as f:
        json.dump(data, f, indent=4)

# Generate and save the data
generate_data()
